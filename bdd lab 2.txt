--Alphabetical list (by surname, name) of all clients with active contract in the date when the query is
-- run, along with start and end dates and type (PPC/PPV) of the contract.

SELECT DISCTINCT surname, name
from clients
join 
join
where WHERE (sysdate >= startdate) AND (sysdate <= enddate)
ORDER BY surname


SELECT DISCTINCT startdate, enddate, contract_type 
FROM contracts
where WHERE (sysdate >= startdate) AND (sysdate <= enddate)
