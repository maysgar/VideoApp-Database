--c) Clients who have bought license for a complete season of any TVseries (include the title and season).


SELECT client, title, season
FROM lic_series 
WHERE COUNT (episode) = (SELECT episodes FROM seasons)






CREATE VIEW temporada AS
SELECT client, title, season, episode, datetime
FROM lic_series;

CREATE VIEW unaserie AS
SELECT client, title, season, episode, COUNT(datetime) AS num
FROM temporada
GROUP BY season, COUNT(datetime);

CREATE VIEW totalSeason AS
SELECT client, title, season, episode, num
FROM series
JOIN series ON (SERIES.title = UNASERIE.title)
WHERE num = total_seasons;

CREATE VIEW final AS
SELECT client, title, season
FROM totalSeason
