SELECT actor
FROM casts  
GROUP BY actor
HAVING COUNT(title) IN (SELECT (movie_title) FROM movies WHERE country= 'USA') 
ORDER BY COUNT(title) DESC;

CREATE VIEW america AS 
SELECT movie_title
FROM movies
WHERE country = 'USA';

CREATE VIEW five AS
SELECT actor
FROM casts
JOIN 



(
SELECT actor
FROM casts
JOIN movies
ON (MOVIES.movie_title = CASTS.title)
WHERE (country = 'USA') AND (ROWNUM <= 5)
ORDER BY COUNT(movie_title) DESC;
